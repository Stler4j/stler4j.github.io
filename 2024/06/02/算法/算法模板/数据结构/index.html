并查集

class UnionFindSet {
    int groups;
    int[] parent, size;
    UnionFindSet(int n) {
        parent = IntStream.range(0, n).toArray();
        size = IntStream.range(0, n).map(_ -> 1).toArray();
    }
    int find(int x) {
        return parent[x] == x ? x : (parent[x] = find(parent[x]));
    }
    boolean merge(int x, int y) {
        if ((x = find(x)) == (y = find(y))) {
            return false;
        }
        parent[x] = y;
        size[y] += size[x];
        --groups;
        return true;
    }
    int sizeOf(int x) {
        return size[find(x)];
    }
    boolean isLeader(int x) {
        return x == find(x);
    }
    boolean isConnected(int x, int y) {
        return find(x) == find(y);
    }
}

二维并查集

class UnionFindSet2D {
    int m;
    UnionFindSet U;
    UnionFindSet2D(int n, int m) {
        this.m = m;
        U = new UnionFindSet(n * m);
    }
    int flat(int x, int y) {
        return m * x + y;
    }
    int find(int x, int y) {
        return U.find(flat(x, y));
    }
    boolean merge(int x1, int y1, int x2, int y2) {
        return U.merge(flat(x1, y1), flat(x2, y2));
    }
    int sizeOf(int x, int y) {
        return U.sizeOf(flat(x, y));
    }
    boolean isLeader(int x, int y) {
        return U.isLeader(flat(x, y));
    }
    boolean isConnected(int x1, int y1, int x2, int y2) {
        return U.isConnected(flat(x1, y1), flat(x2, y2));
    }
}

树状数组

class FenwickTree {
    long[] sum, isum;
    FenwickTree(int n) {
        sum = new long[n + 5];
        isum = new long[n + 5];
    }
    int lowbit(int x) {
        return x & -x;
    }
    void add(int l, int r, long k) {
        add0(l, k);
        add0(r + 1, -k);
    }
    void add0(int i, long k) {
        for (int j = i; j < sum.length; j += lowbit(j)) {
            sum[j] += k;
            isum[j] += k * i;
        }
    }
    long sum(int l, int r) {
        return sum0(r) - sum0(l - 1);
    }
    long sum0(int i) {
        long res = 0;
        for (int j = i; j > 0; j -= lowbit(j)) {
            res += sum[j] * (i + 1) - isum[j];
        }
        return res;
    }
}

二维树状数组

class FenwickTree2D {
    long[][] t1, t2, t3, t4;
    FenwickTree2D(int n, int m) {
        t1 = new long[n + 5][m + 5];
        t2 = new long[n + 5][m + 5];
        t3 = new long[n + 5][m + 5];
        t4 = new long[n + 5][m + 5];
    }
    int lowbit(int x) {
        return x & -x;
    }
    void add(int x1, int y1, int x2, int y2, long k) {
        add0(x1, y1, k);
        add0(x1, y2 + 1, -k);
        add0(x2 + 1, y1, -k);
        add0(x2 + 1, y2 + 1, k);
    }
    void add0(int x, int y, long k) {
        for (int i = x; i < t1.length; i += lowbit(i)) {
            for (int j = y; j < t1[0].length; j += lowbit(j)) {
                t1[i][j] += k;
                t2[i][j] += k * x;
                t3[i][j] += k * y;
                t4[i][j] += k * x * y;
            }
        }
    }
    long sum(int x1, int y1, int x2, int y2) {
        return sum0(x2, y2) - sum0(x2, y1 - 1) - sum0(x1 - 1, y2) + sum0(x1 - 1, y1 - 1);
    }
    long sum0(int x, int y) {
        long res = 0;
        for (int i = x; i > 0; i -= lowbit(i)) {
            for (int j = y; j > 0; j -= lowbit(j)) {
                res += t1[i][j] * (x + 1) * (y + 1) - t2[i][j] * (y + 1) - t3[i][j] * (x + 1) + t4[i][j];
            }
        }
        return res;
    }
}
